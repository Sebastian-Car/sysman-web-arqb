Ruta:NOMINA/INFORMES/LISTADOS ESPECIALES/INFORMES RESOLUCIONES opci¾n 6.Certificado

TIPO : CERTIFICADO

SQl:SELECT NVL(PCK_SYSMAN_UTL.FC_PAR(PERSONAL.COMPANIA, 'TITULO CODIGO CERTIFICACION', 6,SYSDATE),'CERT - ') PAR_TITULOCODCERT, 
     PCK_SYSMAN_UTL.FC_PAR(PERSONAL.COMPANIA, 'TITULO EN CERTICACIONES LABORALES', 6,SYSDATE) PAR_TITULO_EN_CERT_LABORALES ,
     COMPANIA.NITCOMPANIA NITCOMPANIA,
     NVL(PCK_SYSMAN_UTL.FC_PAR(PERSONAL.COMPANIA, 'TITULO CERTIFICACION LABORAL', 6,SYSDATE),'C E R T I F I C A')PAR_TITULO_CERT_LABORAL,
     PERSONAL.NOMBRECOMPLETO NOMBRECOMPLETO,
     TO_CHAR(PERSONAL.NUMERO_DCTO,'999,999,999,999,999,999,999') NUMERO_DCTO,
     CIUDAD.NOMBRE NOMBRECIUDAD,
     CASE WHEN PERSONAL.ESTADO_ACTUAL IN (3) THEN 'labor¾' ELSE 'labora' END ESTADO,
     s$companiaNombre$s COMPANIA_NOMBRE,
     FORMANOMBRAMIENTO.NOMBREFORMA NOMBREFORMA,
     EXTRACT(DAY FROM PERSONAL.FECHA_DE_INGRESO) DIA_INGRESO,
     PCK_SYSMAN_UTL.FC_NOMBRE_MES(EXTRACT(MONTH FROM PERSONAL.FECHA_DE_INGRESO)) NOMBREMES,
     EXTRACT(YEAR FROM PERSONAL.FECHA_DE_INGRESO) ANIO_INGRESO,
     CASE WHEN PERSONAL.ESTADO_ACTUAL IN (3) THEN 'hasta el ' || EXTRACT(DAY FROM PERSONAL.FECHATERCONTRATO)||' de '||PCK_SYSMAN_UTL.FC_NOMBRE_MES(EXTRACT(MONTH FROM PERSONAL.FECHATERCONTRATO))||' de '||EXTRACT(YEAR FROM PERSONAL.FECHATERCONTRATO) ELSE '' END HASTA,
     CARGOS.NOMBRE_DEL_CARGO NOMBRE_DEL_CARGO,
     PCK_SYSMAN_UTL.FC_VALOR_LETRAS(CATEGORIA.SALARIO_BASE) SALARIOBASE_LETRAS, 
     TO_CHAR(CATEGORIA.SALARIO_BASE,'999,999,999,999,999,999,999.99') SALARIO_BASE,
     PCK_SYSMAN_UTL.FC_PAR(PERSONAL.COMPANIA, 'CIUDAD DONDE SE EXPIDE CERTIFICACION LABORAL', 6,SYSDATE) PAR_CIUDADEXPIDE,     
     TO_CHAR(CASE WHEN s$fechaReporte$s = '0' THEN SYSDATE ELSE TO_DATE(s$fechaReporte$s ,'DD/MM/YYYY') END , 'DD " dĒas del mes de "month" de "YYYY','NLS_DATE_LANGUAGE=SPANISH') FECHAREPORTE_LARGA,    
     PCK_SYSMAN_UTL.FC_PAR(PERSONAL.COMPANIA, 'NOMBRE DE QUIEN FIRMA CERTIFICADO DE SALARIOS', 6,SYSDATE) PAR_NOMBRE_FIRMA_CER_SALARIOS,
     PCK_SYSMAN_UTL.FC_PAR(PERSONAL.COMPANIA, 'CARGO DE QUIEN FIRMA CERTIFICADO DE SALARIOS', 6,SYSDATE) PAR_CARGO_FIRMA_CER_SALARIOS,
     PCK_SYSMAN_UTL.FC_PAR(PERSONAL.COMPANIA, 'ELABORADO POR', 6,SYSDATE) PAR_ELABORADO_POR
 FROM PERSONAL INNER JOIN COMPANIA ON PERSONAL.COMPANIA = COMPANIA.CODIGO
     LEFT JOIN CARGOS ON PERSONAL.COMPANIA = CARGOS.COMPANIA
             AND PERSONAL.ID_DE_CARGO = CARGOS.ID_DE_CARGO
     LEFT JOIN CIUDAD ON PERSONAL.PAIS_CED = CIUDAD.PAIS
              AND PERSONAL.EXPEDIDA = CIUDAD.CODIGO
              AND PERSONAL.DEPARTAMENTO_CED = CIUDAD.DEPARTAMENTO
     LEFT JOIN CATEGORIA ON PERSONAL.COMPANIA = CATEGORIA.COMPANIA
              AND PERSONAL.ANO = CATEGORIA.ANO
              AND PERSONAL.ID_DE_CATEGORIA = CATEGORIA.ID_DE_CATEGORIA
              AND PERSONAL.ESCALAFON = CATEGORIA.ESCALAFON               
     LEFT JOIN ESCALAFON ON PERSONAL.COMPANIA = ESCALAFON.COMPANIA     
              AND PERSONAL.ESCALAFON = ESCALAFON.CODIGO
     LEFT JOIN FORMANOMBRAMIENTO ON PERSONAL.COMPANIA = FORMANOMBRAMIENTO.COMPANIA 
               AND PERSONAL.DE_CARRERA = FORMANOMBRAMIENTO.IDFORMA                              
 WHERE PERSONAL.COMPANIA = s$compania$s
  AND PERSONAL.ID_DE_EMPLEADO = s$idEmpleado$s
