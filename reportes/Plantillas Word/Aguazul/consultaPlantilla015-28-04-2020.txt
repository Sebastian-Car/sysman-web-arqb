SELECT ORDENDECOMPRA.COMPANIA,
ORDENDECOMPRA.CLASEORDEN ,
NOVEDADCONTRATO.NOVEDAD,
ORDENDECOMPRA.NUMERO ,
ORDENDECOMPRA.NUMEROACTA, 
TO_CHAR(ORDENDECOMPRA.FECHA,'DD "de" month "del" YYYY','nls_date_language=spanish') FECHANOVEDAD,
TIPOADJUDICACION.DESCRIPCION TIPOADJUDICACION,
TIPOORDENDECOMPRA.NOMBRE NOMBRECONTRATO,
TO_CHAR(ORDENDECOMPRA.FECHAFIRMA,'DD "de" month "del" YYYY','nls_date_language=spanish') FECHAFIRMA,
ORDENDECOMPRA.NOPROCESO,
ORDENDECOMPRA.OBJETOCONTRATO OBJETO,
TO_CHAR(ORDENDECOMPRA.VALORFINAL,'999,999,999,999,999,999.99') VALORINCIALCONTR,
TO_CHAR((ORDENDECOMPRA.VALORTOTAL-ORDENDECOMPRA.VALORFINAL),'999,999,999,999,999,999.99') ADICIONESCONTRATO,
TO_CHAR(ORDENDECOMPRA.VALORPAGOS,'999,999,999,999,999,999.99') VALORTOTALEJECUTADO,
TO_CHAR(ORDENDECOMPRA.VALORTOTAL,'999,999,999,999,999,999.99')  TOTAL,
TO_CHAR(NOVEDADCONTRATO.VALOR_LIBERADO,'999,999,999,999,999,999.99')  SALDOPORPAGAR,
TO_CHAR(NOVEDADCONTRATO.VALORTOTAL,'999,999,999,999,999,999.99')  VALORNOVEDAD,
TERCERO.NOMBRE NOMBRETERCERO,
TERCERO.EXPEDIDACEDULA EXPEDIDATERCERO,
NOVEDADCONTRATO.CONSIDERANDOS,
TO_CHAR(ORDENDECOMPRA.TERCERO,'9G999G999G999','NLS_NUMERIC_CHARACTERS = ".,"') NITERCERO,
PCK_SYSMAN_UTL.FC_DCH(ORDENDECOMPRA.TERCERO)  DVCONTRATISTA,
ORDENDECOMPRA.DURACION,
TO_CHAR(ORDENDECOMPRA.FECHAINICIO,'DD "de" month "del" YYYY','nls_date_language=spanish') FECHAINICIO,
TO_CHAR(ORDENDECOMPRA.FECHAFINAL,'DD "de" month "del" YYYY','nls_date_language=spanish') FECHAFINAL,
TO_CHAR(ORDENDECOMPRA.FECHAFINALIZACION,'DD "de" month "del" YYYY','nls_date_language=spanish') FECHAFECHAFINALIZACION,
TO_CHAR(NOVEDADCONTRATO.FECHAINICIAL,'DD "de" month "del" YYYY','nls_date_language=spanish') FECHAINICIONOVEDAD,
TO_CHAR(NOVEDADCONTRATO.FECHAFINAL,'DD "de" month "del" YYYY','nls_date_language=spanish') FECHAFINALNOVE,
TO_CHAR(ORDENDECOMPRA.FECHAFINALACTUAL, 'DD "de" month "del" YYYY', 'nls_date_language=spanish') FECHAFINALACTUAL,
ORDENDECOMPRA.INTERVENTOR,
TERCERO2.NOMBRE  NOMBSUPERVISOR,
TERCERO2.CARGO  CARGOSUPERVISOR,
TO_CHAR(TERCERO2.NIT,'9G999G999G999','NLS_NUMERIC_CHARACTERS = ".,"') CEDULASUPERVISOR,
TERCERO2.EXPEDIDACEDULA  EXPEDIDASUPERVISOR,
ORDENDECOMPRA.VALOR_ANTICIPO VALORANTICIPO,
ORDENDECOMPRA.PORC_ANTICIPO PORCENTAJEANTICIPO,
NOVEDADCONTRATO.OBSERVACIONES OBSERVACIONESNOVEDAD,
TO_CHAR(ORDENDECOMPRA.FECHAPOLIZAS,'DD "de" month "del" YYYY','nls_date_language=spanish') FECHAPOLIZAS
FROM ORDENDECOMPRA
LEFT JOIN TIPOORDENDECOMPRA 
    ON ORDENDECOMPRA.COMPANIA    = TIPOORDENDECOMPRA.COMPANIA
    AND ORDENDECOMPRA.CLASEORDEN = TIPOORDENDECOMPRA.CODIGO
 LEFT JOIN NOVEDADCONTRATO 
   ON ORDENDECOMPRA.COMPANIA     =  NOVEDADCONTRATO.COMPANIA
  AND ORDENDECOMPRA.CLASEORDEN   =  NOVEDADCONTRATO.CLASEORDEN
  AND ORDENDECOMPRA.NUMERO       =  NOVEDADCONTRATO.ORDENDECOMPRA
 LEFT JOIN ORDENADOR
  ON ORDENDECOMPRA.COMPANIA      =  ORDENADOR.COMPANIA
  AND ORDENDECOMPRA.ORDENADOR    =  ORDENADOR.CEDULA
 LEFT JOIN DEPENDENCIA
  ON ORDENDECOMPRA.COMPANIA                =  DEPENDENCIA.COMPANIA
  AND ORDENDECOMPRA.DEPENDENCIA            =  DEPENDENCIA.CODIGO
 LEFT JOIN SECTORES
  ON ORDENDECOMPRA.COMPANIA                =  SECTORES.COMPANIA
  AND ORDENDECOMPRA.SECTOR                 =  SECTORES.CODIGO
 LEFT JOIN TERCERO 
  ON ORDENDECOMPRA.COMPANIA               =  TERCERO.COMPANIA  
  AND ORDENDECOMPRA.TERCERO                 =  TERCERO.NIT
 LEFT JOIN DEPARTAMENTO
  ON TERCERO.PAIS                          =  DEPARTAMENTO.PAIS
  AND TERCERO.DEPARTAMENTO                 =  DEPARTAMENTO.CODIGO
 LEFT JOIN CIUDAD
  ON TERCERO.PAIS                          =  CIUDAD.PAIS
  AND TERCERO.DEPARTAMENTO                 =  CIUDAD.DEPARTAMENTO
  AND TERCERO.CIUDAD                       =  CIUDAD.CODIGO
 LEFT JOIN TRANSACCION
  ON ORDENDECOMPRA.COMPANIA                =  TRANSACCION.COMPANIA
  AND ORDENDECOMPRA.NOPROCESO              =  TRANSACCION.NUM_PROCESO
 LEFT JOIN ES_ESTPREVIO
  ON TRANSACCION.COMPANIA                  =  ES_ESTPREVIO.COMPANIA
  AND TRANSACCION.ESTUDIOPREVIO            =  ES_ESTPREVIO.COD_ESTUDIO
 LEFT JOIN ES_EST_PROY
  ON ES_ESTPREVIO.COMPANIA                 =  ES_EST_PROY.COMPANIA
  AND ES_ESTPREVIO.COD_ESTUDIO             =  ES_EST_PROY.COD_ESTUDIO
 LEFT JOIN PROYECTOS
  ON ES_EST_PROY.COMPANIA                  =  PROYECTOS.COMPANIA
  AND ES_EST_PROY.COD_PROY                 =  PROYECTOS.CODIGO
 LEFT JOIN TIPOADJUDICACION
  ON ORDENDECOMPRA.COMPANIA                =  TIPOADJUDICACION.COMPANIA
  AND ORDENDECOMPRA.TIPOADJUDICACION       =  TIPOADJUDICACION.CODIGO
 LEFT JOIN TIPOORDENDECOMPRA
  ON ORDENDECOMPRA.COMPANIA                =  TIPOORDENDECOMPRA.COMPANIA
  AND ORDENDECOMPRA.CLASEORDEN             =  TIPOORDENDECOMPRA.CODIGO
 LEFT JOIN SUPERVISORES
  ON ORDENDECOMPRA.COMPANIA                =  SUPERVISORES.COMPANIA
  AND ORDENDECOMPRA.CLASEORDEN             =  SUPERVISORES.CLASEORDEN
  AND ORDENDECOMPRA.NUMERO                 =  SUPERVISORES.NUMEROCONTRATO
 LEFT JOIN TERCERO  TERCERO2
  ON SUPERVISORES.COMPANIA                 =  TERCERO2.COMPANIA
  AND SUPERVISORES.CEDULA                  =  TERCERO2.NIT
 LEFT JOIN V_ORDENDECOMPRA_DIS
  ON ORDENDECOMPRA.COMPANIA                =  V_ORDENDECOMPRA_DIS.COMPANIA
  AND ORDENDECOMPRA.CLASEORDEN             =  V_ORDENDECOMPRA_DIS.CLASEORDEN
  AND ORDENDECOMPRA.NUMERO                 =  V_ORDENDECOMPRA_DIS.NUMERO
 WHERE ORDENDECOMPRA.COMPANIA      = s$compania$s
  AND ORDENDECOMPRA.CLASEORDEN    = TO_CHAR(s$CLASEORDEN$s)
  AND ORDENDECOMPRA.NUMERO      BETWEEN s$NUMERO$s AND s$NUMERO$s
  AND NOVEDADCONTRATO.NOVEDAD  =  s$NOVEDAD$s
