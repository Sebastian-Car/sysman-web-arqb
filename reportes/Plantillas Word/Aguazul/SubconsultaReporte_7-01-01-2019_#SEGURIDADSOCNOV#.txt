SELECT TO_CHAR(S.CODIGOPLANI) "No PLANILLA", TO_CHAR(S.PERIODO) PERIODO ,     TO_CHAR(S.IBC, '999,999,999,999,999,999.99') SIBC, TO_CHAR(S.SALUD, '999,999,999,999,999,999.99') SALUD,  TO_CHAR(S.PENSION, '999,999,999,999,999,999.99') PENSION,   TO_CHAR(S.ARL, '999,999,999,999,999,999.99') SARL,  TO_CHAR(S.CAJACOMPEN, '999,999,999,999,999,999.99') "CAJA COMPENSACION", TO_CHAR(S.SENA, '999,999,999,999,999,999.99') SENA,           TO_CHAR(S.ICBF, '999,999,999,999,999,999.99') ICBF, TO_CHAR(S.TOTALAPORTES, '999,999,999,999,999,999.99') "VALOR TOTAL APORTES"  FROM    SEGURIDADSOCIALNOV   S WHERE s$CONDICIONENLACE$s   UNION  SELECT 'TOTALES' "No PLANILLA", ''  PERIODO,   TO_CHAR(Sum(S.IBC), '999,999,999,999,999,999.99') IBC, TO_CHAR(Sum(S.SALUD), '999,999,999,999,999,999.99')  SSALUD, TO_CHAR(Sum(S.PENSION), '999,999,999,999,999,999.99') AS SPENSION,   TO_CHAR(Sum(S.ARL), '999,999,999,999,999,999.99')  ARL, TO_CHAR(Sum(S.CAJACOMPEN), '999,999,999,999,999,999.99')  SCAJACOMPEN, TO_CHAR(Sum(S.SENA), '999,999,999,999,999,999.99')  SSENA,            TO_CHAR(Sum(S.ICBF), '999,999,999,999,999,999.99')  SICBF, TO_CHAR(Sum(S.TOTALAPORTES), '999,999,999,999,999,999.99')  "VALOR TOTAL APORTES"     FROM    SEGURIDADSOCIALNOV   S  GROUP BY S.COMPANIA, S.CLASEORDEN, S.ORDENDECOMPRA, S.NOVEDAD    HAVING s$CONDICIONENLACE$s
