SELECT * FROM(
SELECT 
E.COMPANIA, 
E.COD_ESTUDIO, 
E.NOMBRE_ESTUDIO,
DEPENDENCIA.NOMBRE DEPENDENCIA,
TERCERO.NOMBRE RESPONSABLE,
DEPENDENCIA_RESPONSABLE.CARGO CARGO_RESPONSABLE,
A.COD_SECTOR || A.CONSECUTIVO AS COD,
A.ADJUNTO
FROM ES_ESTPREVIO E
INNER JOIN ES_ADJUNTOS_ESTPREVIO A
    ON A.COMPANIA = E.COMPANIA
    AND A.COD_ESTUDIO = E.COD_ESTUDIO
LEFT JOIN DEPENDENCIA
    ON E.COMPANIA     = DEPENDENCIA.COMPANIA
    AND E.DEPENDENCIA = DEPENDENCIA.CODIGO
LEFT JOIN DEPENDENCIA_RESPONSABLE
    ON E.COMPANIA     = DEPENDENCIA_RESPONSABLE.COMPANIA
    AND E.DEPENDENCIA = DEPENDENCIA_RESPONSABLE.DEPENDENCIA
    AND E.RESPONSABLE = DEPENDENCIA_RESPONSABLE.RESPONSABLE
LEFT JOIN TERCERO
    ON DEPENDENCIA_RESPONSABLE.RESPONSABLE = TERCERO.NIT
    AND DEPENDENCIA_RESPONSABLE.COMPANIA   = TERCERO.COMPANIA
WHERE E.COMPANIA = s$compania$s
AND E.COD_ESTUDIO = s$codEstudio$s
)PIVOT (
  MAX(ADJUNTO) FOR COD IN (
    'IMG_ASP_GEN1' IMG_ASP_GEN1,
    'IMG_ASP_GEN2' IMG_ASP_GEN2,
    'IMG_ASP_GEN3' IMG_ASP_GEN3,
    'IMG_ASP_ECO1' IMG_ASP_ECO1,
    'IMG_ASP_ECO2' IMG_ASP_ECO2
  )
)
